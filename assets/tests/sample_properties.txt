# Sample CTL Properties for Testing
# This file demonstrates various CTL formulas that can be analyzed

# Basic atomic propositions (Class 1)
p
q
r

# Simple boolean combinations (Class 1)
p & q
p | q
p -> q
!p
!(p & q)

# Complex boolean formulas (Class 1)
(p & q) | r
p & (q | r)
(p -> q) & (q -> r)
p -> (q & r)

# Basic temporal operators (Class 2)  
EF ready
AF ready
EG ready
AG ready

# Temporal with boolean combinations (Class 2)
EF (ready & !error)
AG (ready -> !error)
AF (ready | error)
EG (ready & safe)

# Until operators (Class 2)
E(init U ready)
A(init U ready)
E(trying U critical)
A(trying U critical)

# Complex temporal formulas (Class 2)
AG (request -> AF grant)
AG (trying -> AF critical)
AG (critical -> EF !critical)
EF (p & AG q)
AF (EG p | EF q)

# Arithmetic comparisons (Class 4)
x > 0
y <= 10
t == 5
counter != 0

# Temporal with arithmetic (Class 4)
AG (x > 0 -> AF (x == 0))
EF (counter > 10)
AG (t <= 100)
AF (y >= target)

# Complex mixed formulas (Class 4)
AG ((x > 0 & ready) -> AF (x == 0 | error))
EF (counter > 5 & ready & !error)
